{
    "rules": {        
        "users": {
            "$user_id": {
                ".read": "auth != null && auth.uid == $user_id",
                ".write": "auth != null && auth.uid == $user_id"
            }
        },
        "rooms": {
            "$room_id": {
                ".read": "auth != null && 
                    (data.child('/started').val() == false ||
                    data.child('/users/' + auth.uid).exists())",
                ".write": "auth != null && 
                    ((!data.exists() || data.child('started').val() == false) ||
                    root.child('rooms/' + $room_id + '/users/' + auth.uid).exists())"
            }
        }
    }
}
